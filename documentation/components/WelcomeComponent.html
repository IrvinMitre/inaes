import { Product, ProductModel } from './../../models/productRequest.model';
import * as ProductActions from './../../actions/product.actions';
import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { DirectoryService } from './../../services/directory.service';
import { Store } from '@ngrx/store';
import { _countGroupLabelsBeforeOption } from '@angular/material/core';
import { Router } from '@angular/router';
@Component({
  selector: 'ess-welcome',
  templateUrl: './welcome.component.html',
  styleUrls: ['./welcome.component.scss'],
})
export class WelcomeComponent implements OnInit {
  constructor(private store: Store<{ product: ProductModel }>, private fb: FormBuilder, private dirServ: DirectoryService, private router: Router) { }
  postForm: FormGroup;
  public imagePath;
  imgURL: any;
  public message: string;
  lat = 0;
  lng = 0;
  zoom = 8;
  center = { lat: 0, lng: 0 }
  url = '../../../assets/image.png';
  url2 = '../../../assets/image.png';
  url3 = '../../../assets/image.png';
  empresas: any[];
  nombreEmpresa: any;
  tipo: any;
  referencia: any;
  latitud: any;
  longitud: any;
  ngOnInit() {
    this.postForm = this.fb.group({
      empresa: ['', [Validators.required]],
      tipo: ['', [Validators.required]],
      direccion: ['', [Validators.required]],
      referencia: ['', [Validators.required]],
      img1: [null],
      img2: [null],
      img3: [null],
    });
    this.getDatosOrganismoPlataforma("884", 10);
    this.store.dispatch(ProductActions.LoadProduct());
  }

  onSelectFile(event) {
    if (event.target.files && event.target.files[0]) {
      var reader = new FileReader();
      reader.readAsDataURL(event.target.files[0]); // read file as data url
      reader.onload = (event: any) => { // called once readAsDataURL is completed
        this.url = event.target.result;
      }
    }
  }
  onSelectFile2(event) {
    if (event.target.files && event.target.files[0]) {
      var reader = new FileReader();
      reader.readAsDataURL(event.target.files[0]); // read file as data url
      reader.onload = (event: any) => { // called once readAsDataURL is completed
        this.url2 = event.target.result;
      }
    }
  }
  onSelectFile3(event) {
    if (event.target.files && event.target.files[0]) {
      var reader = new FileReader();
      reader.readAsDataURL(event.target.files[0]); // read file as data url
      reader.onload = (event: any) => { // called once readAsDataURL is completed
        this.url3 = event.target.result;
      }
    }
  }

  getDatosOrganismoPlataforma(idOrganismo, idTipoOrganismo) {
    this.dirServ.getDatosOrganismoPlataforma(idOrganismo, idTipoOrganismo).subscribe(
      res => {
        this.empresas = (res as any).data;
        this.nombreEmpresa = (res as any).data.nombreEmpresa;
        this.tipo = (res as any).data.tipoEmpresa;
        this.referencia = (res as any).data.domicilio.referencia;
        this.latitud = (res as any).data.domicilio.latitud;
        this.longitud = (res as any).data.domicilio.longitud;
        console.log('Empresas', this.empresas);
        this.center = { lat: this.latitud, lng: this.longitud }
        this.postForm = this.fb.group({
          empresa: [this.nombreEmpresa],
          tipo: [this.tipo],
          direccion: [this.referencia],
          referencia: [''],
        });
      },
      err => {
        console.log(err);
      }
    );
  }
  create() {
    window.open("http://sistemas.inaes.gob.mx/directorioEESS/", "_blank"); // set endpoint for creation
  }

  submit() {
    if (!this.postForm) {
      return;
    }
    this.store.dispatch(ProductActions.SetLatLong({ lat: this.latitud, long: this.longitud  }));
    this.router.navigateByUrl('/registro/bienes-y-servicios')
    //console.log(this.postForm.value);

  }
}
