import { Product, ProductModel } from './../../models/productRequest.model';
import * as ProductActions from './../../actions/product.actions';
import { select, Store } from '@ngrx/store';
import { Router } from '@angular/router';
import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import {MatDialog} from '@angular/material/dialog';

@Component({
  selector: 'ess-information',
  templateUrl: './information.component.html',
  styleUrls: ['./information.component.scss']
})
export class InformationComponent implements OnInit {

  constructor(private store: Store<{ product: ProductModel }>,private fb: FormBuilder,  public dialog: MatDialog,private router: Router) { }
  postForm: FormGroup;
  hide = true;
  ngOnInit() {
    this.postForm = this.fb.group({
      correo: [''],
      facebook: [''],
      insta: [''],
      red: [''],
      telefono: [''],
      telefonoW: [''],
    });
  }

  openDialog() {
    const dialogRef = this.dialog.open(ModalDialog4);
  }

  submit() {
    if (!this.postForm.valid) {
      return;
    }
    this.store.dispatch(ProductActions.SetProfile({ website: this.postForm.value.correo, phone: this.postForm.value.telefono, whatsapp: this.postForm.value.telefonoW }));
    this.router.navigateByUrl('/registro/perfil')
    console.log(this.postForm.value);
  }
}
