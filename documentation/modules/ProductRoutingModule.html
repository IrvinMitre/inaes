import { NgModule } from '@angular/core';
import { Routes, RouterModule } from '@angular/router';


import { ProductsComponent } from './products.component';
import { ProductsCategoryComponent } from './../../components/products-category/products-category.component';
import { ProductKeywordComponent } from './../../components/product-keyword/product-keyword.component';
import { ProductDescriptionComponent } from './../../components/product-description/product-description.component';
import { ProductCertComponent } from './../../components/product-cert/product-cert.component';
import { ProductPhotoComponent } from './../../components/product-photo/product-photo.component';
import { ProductEntregaComponent } from './../../components/product-entrega/product-entrega.component';
import { ProductAdicionalComponent } from './../../components/product-adicional/product-adicional.component';
import { ProductPayComponent } from './../../components/product-pay/product-pay.component';
import { ProductPathComponent } from './../../components/product-path/product-path.component';
import { ProductServiceComponent } from './../../components/product-service/product-service.component';
import { ServiciosComponent } from './../../components/servicios/servicios.component';


const routes: Routes = [
  {
    path: '',
    component: ProductsComponent,
    children: [
      {
        path: 'categoria',
        component: ProductsCategoryComponent
      },
      {
        path: 'certificado',
        component: ProductCertComponent
      },
      {
        path: 'descripcion',
        component: ProductDescriptionComponent
      },
      {
        path: 'palabras-clave',
        component: ProductKeywordComponent
      },
      {
        path: 'fotografia',
        component: ProductPhotoComponent
      },
      {
        path: 'entrega',
        component: ProductEntregaComponent
      },
      {
        path: 'adicional',
        component: ProductAdicionalComponent
      },
      {
        path: 'pago',
        component: ProductPayComponent
      },
      {
        path: 'path',
        component: ProductPathComponent
      },
      {
        path: 'producto-servicio',
        component: ProductServiceComponent
      },
      {
        path: 'servicio',
        component: ServiciosComponent
      }


    ]
  }
];

@NgModule({
  imports: [
    RouterModule.forChild(routes)
  ],
  exports: [RouterModule]
})
export class ProductRoutingModule { }
